

** Install libvirt - ubuntu
- install libvirt and related tools
#+BEGIN_SRC sh
apt-get install qemu-kvm libvirt-bin virtinst virt-manager
#+END_SRC

- start and enable the libvirtd
#+BEGIN_SRC sh
systemctl start libvirtd
systemctl enable libvirtd
#+END_SRC
